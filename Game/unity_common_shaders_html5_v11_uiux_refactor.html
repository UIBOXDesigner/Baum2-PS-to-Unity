使用说明

适用时机：普通 UI 元素需要保持贴图原始表现时使用。它是项目 UI 材质库的基准，不承担特效职责。

  1. UGUI 的 Image、Button、RawImage 默认优先使用该材质。
  2. 通过 Image.color 做轻量颜色叠加，不建议为普通按钮额外创建自定义材质。
  3. 需要遮罩时优先确认是否兼容 Mask、RectMask2D 和 Stencil。

参数建议：常用参数为 _MainTex、_Color、Stencil 系列参数。通常不需要暴露额外运行时参数。

注意：不要把所有按钮都换成发光、流光或描边材质，否则会破坏批处理并增加维护成本。

使用方法

脚本调用优先通过 Image.color 控制轻量变色，不建议业务层频繁替换 Material。
Prefab 规范普通按钮 Prefab 使用默认材质；只有特殊状态才额外挂接灰度、描边、流光等材质。

using UnityEngine;
using UnityEngine.UI;

public class UIDefaultColorSetter : MonoBehaviour
{
    [SerializeField] private Image targetImage;

    private void Awake()
    {
        if (targetImage == null)
            targetImage = GetComponent<Image>();
    }

    public void SetColor(Color color)
    {
        if (targetImage == null) return;

        // 普通 UI 不需要自定义材质时，优先使用 Image.color。
        // Material 保持为空时，会走 Unity 默认 UI 材质。
        targetImage.color = color;
    }

    public void ResetColor()
    {
        SetColor(Color.white);
    }
}

UI/Default Shader 脚本示例

将下面代码保存为 UI_DefaultExample.shader，然后创建对应材质并挂到该示例说明的 UI Image 层。

展开 Shader 脚本 
// 说明：以下 Shader 为 UGUI 示例版，适合 Built-in 管线或兼容 CGPROGRAM 的项目参考。
// 如果项目使用 URP，可先验证兼容性；正式项目建议由 TA 根据项目管线统一改造。
Shader "UI/DefaultExample"
{
    Properties
    {
        [PerRendererData] _MainTex ("Sprite Texture", 2D) = "white" {}
        _Color ("Tint", Color) = (1,1,1,1)

        _StencilComp ("Stencil Comparison", Float) = 8
        _Stencil ("Stencil ID", Float) = 0
        _StencilOp ("Stencil Operation", Float) = 0
        _StencilWriteMask ("Stencil Write Mask", Float) = 255
        _StencilReadMask ("Stencil Read Mask", Float) = 255

        _ColorMask ("Color Mask", Float) = 15
        [Toggle(UNITY_UI_ALPHACLIP)] _UseUIAlphaClip ("Use Alpha Clip", Float) = 0
    }

    SubShader
    {
        Tags
        {
            "Queue"="Transparent"
            "IgnoreProjector"="True"
            "RenderType"="Transparent"
            "PreviewType"="Plane"
            "CanUseSpriteAtlas"="True"
        }

        Stencil
        {
            Ref [_Stencil]
            Comp [_StencilComp]
            Pass [_StencilOp]
            ReadMask [_StencilReadMask]
            WriteMask [_StencilWriteMask]
        }

        Cull Off
        Lighting Off
        ZWrite Off
        ZTest [unity_GUIZTestMode]
        Blend SrcAlpha OneMinusSrcAlpha
        ColorMask [_ColorMask]

        Pass
        {
            Name "DefaultExample"

            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_local _ UNITY_UI_CLIP_RECT
            #pragma multi_compile_local _ UNITY_UI_ALPHACLIP

            #include "UnityCG.cginc"
            #include "UnityUI.cginc"

            sampler2D _MainTex;
            fixed4 _Color;
            float4 _ClipRect;

            struct appdata_t
            {
                float4 vertex : POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
                float4 worldPosition : TEXCOORD1;
            };

            v2f vert(appdata_t v)
            {
                v2f o;
                o.worldPosition = v.vertex;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.texcoord = v.texcoord;
                o.color = v.color * _Color;
                return o;
            }

            fixed4 frag(v2f i) : SV_Target
            {
                fixed4 col = tex2D(_MainTex, i.texcoord) * i.color;

                #ifdef UNITY_UI_CLIP_RECT
                col.a *= UnityGet2DClipping(i.worldPosition.xy, _ClipRect);
                #endif

                #ifdef UNITY_UI_ALPHACLIP
                clip(col.a - 0.001);
                #endif

                return col;
            }
            ENDCG
        }
    }
}

使用说明

适用时机：功能未开放、按钮不可点击、道具不可使用、头像锁定、技能不可释放时使用。

  1. 准备一个 Mat_UI_Grayscale 材质，挂到需要置灰的 Image 上。
  2. 运行时通过 _GrayAmount 控制灰度强度；0 表示原色，1 表示完全置灰。
  3. 按钮禁用时同时关闭 Button.interactable，并切换到灰度材质或修改灰度参数。
  4. 如果同一个图标需要频繁切换可用/不可用状态，优先用材质参数控制，不要替换 Sprite。

参数建议：_GrayAmount 建议默认 1；_Alpha 可用于弱化禁用态；_Color 可用于轻微压暗。

注意：不要直接导出灰色图片作为禁用态，否则会增加图集体积，并导致后续换色和皮肤适配成本升高。

使用方法

创建方式创建 Shader "UI/Grayscale"，再创建材质 Mat_UI_Grayscale，默认 _GrayAmount=1。
挂载位置挂到需要置灰的 Image、头像、道具图标、按钮图标上；按钮文字可单独降低透明度或使用 TMP 颜色控制。
脚本调用通过 material.SetFloat("_GrayAmount", value) 控制置灰强度；禁用状态建议同时设置 Button.interactable=false。
Prefab 规范按钮 Prefab 建议保留 NormalMaterial 和 GrayMaterial 两个引用，由状态脚本统一切换。
验收标准禁用态明显但仍可识别原图；恢复可用时颜色正常；不会生成大量独立材质实例。
脚本调用示例（已检查修正） 
using UnityEngine;
using UnityEngine.UI;

public class UIGrayController : MonoBehaviour
{
    [Header("Target")]
    [SerializeField] private Image targetImage;
    [SerializeField] private Button targetButton;

    [Header("Materials")]
    [SerializeField] private Material normalMaterial;
    [SerializeField] private Material grayMaterial;

    private Material runtimeGrayMaterial;

    private void Awake()
    {
        if (targetImage == null)
            targetImage = GetComponent<Image>();

        if (targetButton == null)
            targetButton = GetComponent<Button>();

        if (grayMaterial != null)
        {
            // 每个 UI 独立控制灰度时，必须实例化材质，避免影响其他 UI。
            runtimeGrayMaterial = Instantiate(grayMaterial);
            runtimeGrayMaterial.SetFloat("_GrayAmount", 1f);
        }
    }

    public void SetDisabled(bool disabled)
    {
        if (targetButton != null)
            targetButton.interactable = !disabled;

        if (targetImage != null)
            targetImage.material = disabled ? runtimeGrayMaterial : normalMaterial;

        if (runtimeGrayMaterial != null)
            runtimeGrayMaterial.SetFloat("_GrayAmount", disabled ? 1f : 0f);
    }

    private void OnDestroy()
    {
        if (runtimeGrayMaterial != null)
            Destroy(runtimeGrayMaterial);
    }
}

Shader "UI/Grayscale"
{
    Properties
    {
        [PerRendererData] _MainTex ("Sprite Texture", 2D) = "white" {}
        _Color ("Tint", Color) = (1,1,1,1)
        _GrayAmount ("Gray Amount", Range(0,1)) = 1
        _Alpha ("Alpha", Range(0,1)) = 1

        _StencilComp ("Stencil Comparison", Float) = 8
        _Stencil ("Stencil ID", Float) = 0
        _StencilOp ("Stencil Operation", Float) = 0
        _StencilWriteMask ("Stencil Write Mask", Float) = 255
        _StencilReadMask ("Stencil Read Mask", Float) = 255
        _ColorMask ("Color Mask", Float) = 15
        [Toggle(UNITY_UI_ALPHACLIP)] _UseUIAlphaClip ("Use Alpha Clip", Float) = 0
    }

    SubShader
    {
        Tags
        {
            "Queue"="Transparent"
            "IgnoreProjector"="True"
            "RenderType"="Transparent"
            "PreviewType"="Plane"
            "CanUseSpriteAtlas"="True"
        }

        Stencil
        {
            Ref [_Stencil]
            Comp [_StencilComp]
            Pass [_StencilOp]
            ReadMask [_StencilReadMask]
            WriteMask [_StencilWriteMask]
        }

        Cull Off
        Lighting Off
        ZWrite Off
        ZTest [unity_GUIZTestMode]
        Blend SrcAlpha OneMinusSrcAlpha
        ColorMask [_ColorMask]

        Pass
        {
            Name "Grayscale"

            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_local _ UNITY_UI_CLIP_RECT
            #pragma multi_compile_local _ UNITY_UI_ALPHACLIP

            #include "UnityCG.cginc"
            #include "UnityUI.cginc"

            sampler2D _MainTex;
            fixed4 _Color;
            float _GrayAmount;
            float _Alpha;
            float4 _ClipRect;

            struct appdata_t
            {
                float4 vertex : POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
                float4 worldPosition : TEXCOORD1;
            };

            v2f vert(appdata_t v)
            {
                v2f o;
                o.worldPosition = v.vertex;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.texcoord = v.texcoord;
                o.color = v.color * _Color;
                return o;

            fixed4 frag(v2f i) : SV_Target
            {
                fixed4 col = tex2D(_MainTex, i.texcoord) * i.color;

UI/Outline
图标描边、选中描边、品质框强化。
常用
适用控件奖励图标、角色头像、装备格、选中状态。 接入建议描边采样次数会增加性能开销，移动端避免过厚描边。
_OutlineColor：描边颜色_OutlineWidth：描边宽度_AlphaThreshold：透明边缘阈值

使用说明

适用时机：需要表达选中、可交互、高品质、当前目标、装备格聚焦时使用。

  1. 给奖励图标、装备格、头像框等少量关键元素使用描边材质。
  2. 通过 _OutlineColor 统一配置品质色或选中色。
  3. 通过 _OutlineWidth 控制描边厚度，移动端建议使用较小宽度。
  4. 列表中大量图标需要描边时，优先考虑美术出图或九宫格边框，不要全部实时 Shader 描边。

参数建议：_OutlineColor 控制描边颜色；_OutlineWidth 控制描边宽度；_AlphaThreshold 用于处理透明边缘。

注意：不要在同屏大量元素上使用高采样描边，尤其是背包、邮件、排行榜等密集列表。

UI/Outline 描边 Shader 脚本示例

将下面代码保存为 UI_Outline.shader，然后创建对应材质并挂到该示例说明的 UI Image 层。

展开 Shader 脚本 
// 说明：以下 Shader 为 UGUI 示例版，适合 Built-in 管线或兼容 CGPROGRAM 的项目参考。
// 如果项目使用 URP，可先验证兼容性；正式项目建议由 TA 根据项目管线统一改造。
Shader "UI/Outline"
{
    Properties
    {
        [PerRendererData] _MainTex ("Sprite Texture", 2D) = "white" {}
        _Color ("Tint", Color) = (1,1,1,1)
        _OutlineColor ("Outline Color", Color) = (1,0.85,0.2,1)
        _OutlineWidth ("Outline Width", Range(0,8)) = 2
        _AlphaThreshold ("Alpha Threshold", Range(0,1)) = 0.1

        _StencilComp ("Stencil Comparison", Float) = 8
        _Stencil ("Stencil ID", Float) = 0
        _StencilOp ("Stencil Operation", Float) = 0
        _StencilWriteMask ("Stencil Write Mask", Float) = 255
        _StencilReadMask ("Stencil Read Mask", Float) = 255
        _ColorMask ("Color Mask", Float) = 15
        [Toggle(UNITY_UI_ALPHACLIP)] _UseUIAlphaClip ("Use Alpha Clip", Float) = 0
    }

    SubShader
    {
        Tags
        {
            "Queue"="Transparent"
            "IgnoreProjector"="True"
            "RenderType"="Transparent"
            "PreviewType"="Plane"
            "CanUseSpriteAtlas"="True"
        }

        Stencil
        {
            Ref [_Stencil]
            Comp [_StencilComp]
            Pass [_StencilOp]
            ReadMask [_StencilReadMask]
            WriteMask [_StencilWriteMask]
        }

        Cull Off
        Lighting Off
        ZWrite Off
        ZTest [unity_GUIZTestMode]
        Blend SrcAlpha OneMinusSrcAlpha
        ColorMask [_ColorMask]

        Pass
        {
            Name "Outline"

            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_local _ UNITY_UI_CLIP_RECT
            #pragma multi_compile_local _ UNITY_UI_ALPHACLIP

            #include "UnityCG.cginc"
            #include "UnityUI.cginc"

            sampler2D _MainTex;
            float4 _MainTex_TexelSize;
            fixed4 _Color;
            fixed4 _OutlineColor;
            float _OutlineWidth;
            float _AlphaThreshold;
            float4 _ClipRect;

            struct appdata_t
            {
                float4 vertex : POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
                float4 worldPosition : TEXCOORD1;
            };

            v2f vert(appdata_t v)
            {
                v2f o;
                o.worldPosition = v.vertex;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.texcoord = v.texcoord;
                o.color = v.color * _Color;
                return o;
            }

            fixed4 frag(v2f i) : SV_Target
            {
                fixed4 baseCol = tex2D(_MainTex, i.texcoord) * i.color;

                float2 offset = _MainTex_TexelSize.xy * _OutlineWidth;

                float outlineAlpha = 0;
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(offset.x, 0)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(-offset.x, 0)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(0, offset.y)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(0, -offset.y)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(offset.x, offset.y)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(-offset.x, offset.y)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(offset.x, -offset.y)).a);
                outlineAlpha = max(outlineAlpha, tex2D(_MainTex, i.texcoord + float2(-offset.x, -offset.y)).a);

                float isInner = step(_AlphaThreshold, baseCol.a);
                float isOutline = saturate(outlineAlpha - baseCol.a);

                fixed4 col = lerp(_OutlineColor, baseCol, isInner);
                col.a = max(baseCol.a, isOutline * _OutlineColor.a);

                #ifdef UNITY_UI_CLIP_RECT
                col.a *= UnityGet2DClipping(i.worldPosition.xy, _ClipRect);
                #endif

                #ifdef UNITY_UI_ALPHACLIP
                clip(col.a - 0.001);
                #endif

                return col;
            }
            ENDCG
        }
    }
}

使用说明

适用时机：可领取、稀有奖励、活动入口、限时提示、暴击反馈等需要吸引注意力的 UI 使用。

    [SerializeField] private Image targetImage;

    [Header("Material")]
    [SerializeField] private Material roundedMaterial;

    [Header("Default")]
    [SerializeField] private float defaultRadius = 16f;
    [SerializeField] private float defaultSoftness = 1f;

    private Material runtimeRoundedMaterial;

    private void Awake()
    {
        if (targetImage == null)
            targetImage = GetComponent<Image>();

        if (roundedMaterial != null)
            runtimeRoundedMaterial = Instantiate(roundedMaterial);

        if (targetImage != null)
            targetImage.material = runtimeRoundedMaterial;

        Apply(defaultRadius, defaultSoftness);
    }

    public void SetRadius(float radius)
    {
        defaultRadius = radius;
        Apply(defaultRadius, defaultSoftness);
    }

    public void SetSoftness(float softness)
    {
        defaultSoftness = softness;
        Apply(defaultRadius, defaultSoftness);
    }

    private void Apply(float radius, float softness)
    {
        if (runtimeRoundedMaterial == null) return;

        runtimeRoundedMaterial.SetFloat("_Radius", radius);
        runtimeRoundedMaterial.SetFloat("_Softness", softness);
    }

    private void OnDestroy()
    {
        if (runtimeRoundedMaterial != null)
            Destroy(runtimeRoundedMaterial);
    }
}

UI/RoundedClip 圆角裁剪 Shader 脚本示例

将下面代码保存为 UI_RoundedClip.shader，然后创建对应材质并挂到该示例说明的 UI Image 层。

展开 Shader 脚本 
// 说明：以下 Shader 为 UGUI 示例版，适合 Built-in 管线或兼容 CGPROGRAM 的项目参考。
// 如果项目使用 URP，可先验证兼容性；正式项目建议由 TA 根据项目管线统一改造。
Shader "UI/RoundedClip"
{
    Properties
    {
        [PerRendererData] _MainTex ("Sprite Texture", 2D) = "white" {}
        _Color ("Tint", Color) = (1,1,1,1)
        _Radius ("Radius", Range(0,0.5)) = 0.12
        _Softness ("Softness", Range(0.001,0.2)) = 0.02

        _StencilComp ("Stencil Comparison", Float) = 8
        _Stencil ("Stencil ID", Float) = 0
        _StencilOp ("Stencil Operation", Float) = 0
        _StencilWriteMask ("Stencil Write Mask", Float) = 255
        _StencilReadMask ("Stencil Read Mask", Float) = 255
        _ColorMask ("Color Mask", Float) = 15
        [Toggle(UNITY_UI_ALPHACLIP)] _UseUIAlphaClip ("Use Alpha Clip", Float) = 0
    }

    SubShader
    {
        Tags
        {
            "Queue"="Transparent"
            "IgnoreProjector"="True"
            "RenderType"="Transparent"
            "PreviewType"="Plane"
            "CanUseSpriteAtlas"="True"
        }

        Stencil
        {
            Ref [_Stencil]
            Comp [_StencilComp]
            Pass [_StencilOp]
            ReadMask [_StencilReadMask]
            WriteMask [_StencilWriteMask]
        }

        Cull Off
        Lighting Off
        ZWrite Off
        ZTest [unity_GUIZTestMode]
        Blend SrcAlpha OneMinusSrcAlpha
        ColorMask [_ColorMask]

        Pass
        {
            Name "RoundedClip"

            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_local _ UNITY_UI_CLIP_RECT
            #pragma multi_compile_local _ UNITY_UI_ALPHACLIP

            #include "UnityCG.cginc"
            #include "UnityUI.cginc"

            sampler2D _MainTex;
            fixed4 _Color;
            float _Radius;
            float _Softness;
            float4 _ClipRect;

            struct appdata_t
            {
                float4 vertex : POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
                float4 worldPosition : TEXCOORD1;
            };

            float RoundedBoxMask(float2 uv, float radius, float softness)
            {
                float2 p = abs(uv - 0.5);
                float2 halfSize = float2(0.5 - radius, 0.5 - radius);
                float2 q = p - halfSize;
                float dist = length(max(q, 0.0)) + min(max(q.x, q.y), 0.0) - radius;
                return 1.0 - smoothstep(0.0, softness, dist);
            }

            v2f vert(appdata_t v)
            {
                v2f o;
                o.worldPosition = v.vertex;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.texcoord = v.texcoord;
                o.color = v.color * _Color;
                return o;
            }

            fixed4 frag(v2f i) : SV_Target
            {
                fixed4 col = tex2D(_MainTex, i.texcoord) * i.color;

                float mask = RoundedBoxMask(i.texcoord, _Radius, _Softness);
                col.a *= mask;

                #ifdef UNITY_UI_CLIP_RECT
                col.a *= UnityGet2DClipping(i.worldPosition.xy, _ClipRect);
                #endif

                #ifdef UNITY_UI_ALPHACLIP
                clip(col.a - 0.001);
                #endif

                return col;
            }
            ENDCG
        }
    }
}

Particles/Alpha Blended 必备 四、Particle / 特效类 Shader 普通半透明粒子。 烟雾、雾气、气流、沙尘、软光。 半透明排序和 Overdraw 是主要性能风险。
Particles/SoftAdditive 常用 四、Particle / 特效类 Shader 柔和加法光效。 按钮高光、奖励光晕、温和技能特效。 比强 Additive 更适合 UI 和卡通风格。
Particles/Distortion 可选 四、Particle / 特效类 Shader 热浪、冲击波、空间扭曲。 爆炸冲击、Boss 技能、传送门、热浪。 成本相对较高，建议只用于关键技能。
Terrain/Lit 常用 五、场景与特殊用途 Shader 地形贴图混合、基础地表光照。 大地图、战斗地图、荒野地表、草地。 贴图层数不要过多，移动端注意采样成本。
Water/Simple 可选 五、场景与特殊用途 Shader 简单水面、流动水纹。 河流、湖面、水池、UI 水波背景。 优先使用简单 UV 流动方案，避免复杂实时反射。
Skybox/Procedural 或 Cubemap 常用 五、场景与特殊用途 Shader 天空盒、环境背景。 战斗背景、营地、荒野远景、昼夜变化。 移动端多用静态 Cubemap 或低成本渐变天空。
Stencil/Mask 常用 五、场景与特殊用途 Shader 模板遮罩、局部显示、特殊裁剪。 头像遮罩、窗口裁剪、局部特效限制。 适合复杂 UI 裁剪，但要统一规范 Stencil Ref。

项目接入规范建议

  • Shader 命名建议使用统一前缀，例如 UI/、Sprite/、Model/、Particles/、Scene/。
  • 材质命名建议采用 Mat_模块_用途_状态，例如 Mat_UI_Button_Gray。
  • UI 置灰、流光、冷却遮罩应通过材质参数控制，避免运行时频繁替换图片。
  • 移动端慎用大面积半透明、复杂描边、多层发光和全屏后处理。
  • 粒子类 Shader 需要重点关注 Overdraw，尤其是烟雾、光晕和全屏爆闪。
  • 核心角色可使用 URP/Lit，普通模型优先使用 URP/Simple Lit 或 URP/Unlit。
  • 可配置参数应暴露给材质面板，例如灰度强度、流光速度、溶解阈值、边缘颜色。
  • 所有自定义 Shader 应准备默认材质、示例 Prefab 和测试场景，方便美术验收。
  • UI 自定义 Shader 要检查是否兼容 Mask、RectMask2D、CanvasRenderer、SpriteAtlas、图集 Alpha 通道。
  • 同屏大量 UI 元素尽量复用材质实例，避免为每个控件独立创建 Material 导致批处理被打断。

UI 灰度 Shader 示例

该示例适合 UGUI Image 置灰。正式项目中建议结合项目管线、Mask、RectMask2D、SpriteAtlas、批处理规则再做适配。

Shader "UI/Grayscale"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        _GrayAmount ("Gray Amount", Range(0,1)) = 1
        _Alpha ("Alpha", Range(0,1)) = 1
    }

    SubShader
    {
        Tags
        {
            "Queue"="Transparent"
            "IgnoreProjector"="True"
            "RenderType"="Transparent"
            "PreviewType"="Plane"
            "CanUseSpriteAtlas"="True"
        }

        Cull Off
        Lighting Off
        ZWrite Off
        ZTest [unity_GUIZTestMode]
        Blend SrcAlpha OneMinusSrcAlpha

        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"

            sampler2D _MainTex;
            float4 _MainTex_ST;
            float _GrayAmount;
            float _Alpha;

            struct appdata_t
            {
                float4 vertex : POSITION;
                float4 color : COLOR;
                float2 texcoord : TEXCOORD0;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                fixed4 color : COLOR;
                float2 texcoord : TEXCOORD0;
            };

            v2f vert(appdata_t v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.texcoord = TRANSFORM_TEX(v.texcoord, _MainTex);
                o.color = v.color;
                return o;
            }

            fixed4 frag(v2f i) : SV_Target
            {
                fixed4 col = tex2D(_MainTex, i.texcoord) * i.color;
                float gray = dot(col.rgb, float3(0.299, 0.587, 0.114));
                col.rgb = lerp(col.rgb, gray.xxx, _GrayAmount);
                col.a *= _Alpha;
                return col;
            }
            ENDCG
        }
    }
}

使用建议：先建立基础材质库，再按系统逐步扩展。不要在项目早期一次性接入过多特殊 Shader，否则会增加材质维护、批处理、性能排查和美术规范成本。
